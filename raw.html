int Button1 = 2;
int Button2 = 3;
int Button3 = 4;

int a = 8;
int b = 6;
int c = 7;
int d = 10;
int e = 9;
int g = 5;

void setup()
{
//Inputs
pinMode(Button1, INPUT);
pinMode(Button2, INPUT);
pinMode(Button3, INPUT);
//Outputs
pinMode(a, OUTPUT);
pinMode(b, OUTPUT);
pinMode(c, OUTPUT);
pinMode(d, OUTPUT);
pinMode(e, OUTPUT);
pinMode(g, OUTPUT);
}

void loop()
{
int x,y,z;
x = digitalRead(Button1);
y = digitalRead(Button2);
z = digitalRead(Button3);

if(x == 1){

digitalWrite(b, HIGH);
digitalWrite(c, HIGH);
digitalWrite(e, LOW);
digitalWrite(d, LOW);
digitalWrite(g, LOW);
digitalWrite(a, LOW);
}
else if(y == 1){

digitalWrite(a, HIGH);
digitalWrite(b, HIGH);
digitalWrite(e, HIGH);
digitalWrite(d, HIGH);
digitalWrite(g, HIGH);
digitalWrite(c, LOW);
}
else if(z == 1){

digitalWrite(c, HIGH);
digitalWrite(a, HIGH);
digitalWrite(b, HIGH);
digitalWrite(g, HIGH);
digitalWrite(d, HIGH);
digitalWrite(e, LOW);
}
else{

digitalWrite(c, LOW);
digitalWrite(a, LOW);
digitalWrite(b, LOW);
digitalWrite(g, LOW);
digitalWrite(d, LOW);
digitalWrite(e, LOW);
}
}
